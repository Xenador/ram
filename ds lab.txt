#include<stdio.h>
#include<conio.h>

void main()
{
    
    int array[50],size,i,ele,pos;
    
    printf("enter the size of array");
    
    scanf("%d",&size);
    
    printf("enter the array elements");
    
    for(i=0;i<size;i++)
    {
        
        scanf("%d",&array[i]);
        
    }
    
    printf("enter the position of insertion of element in that array");
    
    scanf("%d",&pos);
    
    printf("enter the element");
    
    scanf("%d",&ele);
    
    for(i=size-1;i>=pos-1;i--)
    {
        
        array[i+1] = array[i];
        
    }
    
    array[pos-1] = ele;
    
    for(i=0;i<size+1;i++)
    {
        
        printf("%d",array[i]);
        
    }
}

    for(i=0;i<n;i++)
    {
        
        scanf("%d",&a[i]);
        
    }
    
    s = (n*(n+1))/2;
